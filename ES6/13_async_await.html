        console.log(error);
    }

try ~ catch 에러 처리

async / await에서는 Promise의 실패를 try ~ catch로 처리합니다.

await 중 실패하면 catch로 이동
function getUser(success) {
    return new Promise((resolve, reject) => {
        if (success) {
            resolve({ id: 1, username: "kosmo" });
        } else {
            reject("사용자 조회 실패");
        }
    });
}

async function loadUser() {
    try {
        const user = await getUser(false);
        console.log(user);
    } catch (error) {
        console.log("에러:", error);
    }
}

loadUser();
// 에러: 사용자 조회 실패

병렬 실행

서로 의존하지 않는 요청이라면 동시에 시작하는 것이 더 효율적입니다.

비효율적 순차 실행 
async function loadDashboard() {
    const users = await fetchUsers();
    const posts = await fetchPosts();
    const comments = await fetchComments();

    console.log(users, posts, comments);

병렬 실행 
async function loadDashboard() {
    try {
        const [users, posts, comments] = await Promise.all([
            fetchUsers(),
            fetchPosts(),
            fetchComments()
        ]);

        console.log(users, posts, comments);
    } catch (error) {
        console.log(error);
    }
}

일부 실패도 확인해야 한다면 allSettled

Promise.allSettled와 await
async function loadAllSettled() {
    const results = await Promise.allSettled([
        fetchUsers(),
        fetchPosts(),
        fetchComments()
    ]);

    results.forEach((result) => {
        if (result.status === "fulfilled") {
            console.log("성공:", result.value);
        } else {
            console.log("실패:", result.reason);
        }
    });
}
상황 추천 방식 이유
앞 결과가 다음 요청에 필요 await 순차 실행 순서를 지켜야 함
요청들이 서로 독립적 await Promise.all() 동시에 요청해 더 효율적
일부 실패도 결과로 확인 await Promise.allSettled() 성공·실패 결과를 모두 확인 가능

loading, error, data 상태 처리

UserList.jsx
import { useEffect, useState } from "react";

function UserList() {
    const [users, setUsers] = useState([]);
    const [loading, setLoading] = useState(false);
    const [error, setError] = useState("");

    useEffect(() => {
        async function loadUsers() {
            try {
                setLoading(true);
                setError("");

                const response = await fetch("/api/users");

                if (!response.ok) {
                    throw new Error("사용자 목록을 불러오지 못했습니다.");
                }

                const data = await response.json();

                setUsers(data);
            } catch (error) {
                setError(error.message);
            } finally {
                setLoading(false);
            }
        }

        loadUsers();
    }, []);

    if (loading) {
        return <p>로딩 중...</p>;
    }

    if (error) {
        return <p>{error}</p>;
    }

    return (
        <ul>
            {users.map((user) => (
                <li key={user.id}>
                    {user.username}
                </li>
            ))}
        </ul>
    );
}

export default UserList;

    });

    if (!response.ok) {
        throw new Error("로그인 실패");
    }

    return await response.json();
}
handleLogin 함수
async function handleLogin() {
    try {
        setLoading(true);
        setError("");

        const data = await login(form.username, form.password);

        localStorage.setItem("accessToken", data.token);

        setError(error.message);

    }

Promise.all과 await
async function loadDashboard() {
    try {
        setLoading(true);

        const [users, posts, comments] = await Promise.all([
            fetch("/api/users").then((res) => res.json()),
            fetch("/api/posts").then((res) => res.json()),
            fetch("/api/comments").then((res) => res.json())
        ]);

        setUsers(users);
        setPosts(posts);
        setComments(comments);
    } catch (error) {
        setError("대시보드 데이터를 불러오지 못했습니다.");
    } finally {
        setLoading(false);
    }
}

02

await를 async 함수 밖에서 사용한다

학습 기준으로 await는 async 함수 내부에서 사용해야 합니다.

03

Promise를 반환하지 않는 함수에 await를 붙인다

function task() {
    setTimeout(() => {
        console.log("완료");
    }, 1000);
}

await task();

task가 Promise를 반환하지 않으면 await가 기다리지 못합니다.

04

try ~ catch 없이 await를 사용한다

async function loadUser() {
    const response = await fetch("/api/users");
    const data = await response.json();

    console.log(data);

요청 실패를 처리하기 어렵습니다.

05

fetch에서 response.ok를 확인하지 않는다

const response = await fetch("/api/wrong-url");
const data = await response.json();

독립 요청을 모두 순차 await로 처리한다

const users = await fetchUsers();
const posts = await fetchPosts();
const comments = await fetchComments();

서로 독립적이면 Promise.all로 병렬 처리하는 것이 효율적입니다.

07

useEffect 콜백을 바로 async로 만든다

useEffect(async () => {
    const data = await fetchData();
    setData(data);
}, []);

내부에 async 함수를 선언하고 호출하는 방식을 권장합니다.

08

finally에서 loading을 끄지 않는다

try {
    setLoading(true);
    const data = await fetchData();
    setData(data);

    setError(error.message);

성공·실패와 관계없이 loading을 false로 돌려야 합니다.

면접 포인트

Q1. async / await란?

Promise 기반 비동기 코드를 동기 코드처럼
위에서 아래로 읽기 쉽게 작성하게 해 주는 문법입니다.

Q2. async 함수의 특징은?

async 함수는 항상 Promise를 반환합니다.
return 값은 fulfilled 값이 되고, throw는 rejected 상태가 됩니다.

Q3. await의 역할은?

Promise가 처리될 때까지 async 함수 내부의 다음 줄 실행을 기다리고,
fulfilled 값은 변수에 담아 사용할 수 있습니다.

Q4. await는 어디에서 사용하나요?

학습 기준으로 async 함수 내부에서 사용합니다.
일반 함수에서 사용하면 SyntaxError가 발생할 수 있습니다.

Q5. async / await에서 에러 처리는?

try ~ catch를 사용합니다. await 중 Promise가 reject되면 catch로 이동합니다.

Q6. await를 여러 번 쓰면 항상 좋은가요?

아닙니다. 의존 관계가 있으면 순차 await가 맞고,
독립 작업이면 Promise.all로 병렬 처리하는 것이 효율적입니다.

Q7. Promise 방식과 차이는?

Promise 방식은 then/catch 체이닝을 사용하고,
async / await는 try/catch와 await를 사용해 더 동기 코드처럼 작성합니다.

Q8. fetch에서 response.ok를 확인하는 이유는?

HTTP 404, 500 응답이 자동으로 catch로 가지 않을 수 있으므로
직접 실패를 throw하기 위해서입니다.

Q9. React useEffect에서는 어떻게 쓰나요?

useEffect 콜백 자체를 async로 만들지 않고,
내부에 async 함수를 선언한 뒤 호출하는 방식이 권장됩니다.

Q10. React에서 어디에 사용되나요?

서버 요청, 로그인 요청, 게시글 목록 조회, 로딩 상태, 에러 상태, 성공 데이터 저장에 사용됩니다.

한 문장으로 마무리

async / await는 Promise 기반 비동기 코드를 동기 코드처럼 위에서 아래로 읽기 쉽게 작성하게 해 주는 문법이며,
async 함수는 항상 Promise를 반환하고, await는 Promise 결과를 기다린 뒤 resolve 값을 꺼냅니다.
실패 처리는 try ~ catch, 마무리는 finally로 처리하며, React에서는 서버 요청을 처리할 때 data, loading, error 상태와 함께 사용합니다.

학습 메모

async / await는 Promise를 더 쉽게 읽기 위한 문법입니다. 따라서 Promise의 resolve, reject, then, catch 흐름을 이해하고 있으면 훨씬 쉽게 익힐 수 있습니다.

async 기준

async function은 항상 Promise를 반환합니다. 함수 내부에서 값을 return해도 호출 결과는 Promise입니다.

await 기준

await는 Promise가 처리될 때까지 기다린 뒤 resolve 값을 변수에 담습니다. 기다리게 하려면 대상 함수가 Promise를 반환해야 합니다.

에러 처리 기준

await 중 reject나 throw가 발생하면 바로 catch로 이동합니다. API 요청 코드는 try ~ catch ~ finally 구조로 작성하는 습관이 좋습니다.

React 기준

useEffect 콜백 자체를 async로 만들지 말고, 내부에 async 함수를 선언한 뒤 호출합니다. 요청 시작 시 loading을 true로, finally에서 false로 돌리는 흐름을 기억합니다.
